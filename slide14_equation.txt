% Feasible set (put below the argmin, or on the next line):

\mathcal{T}_z \;=\; \{\, t \in [0,1] \;:\; F_{1,z}(t) \,\ge\, \alpha \, F_{1,z}(t_{\mathrm{ref}}) \,\}

t_z^{*} \;=\; \arg\min_{t \in \mathcal{T}_z} \,\bigl|\, \mathrm{FPR}_z(t) \,-\, \mathrm{FPR}_{\mathrm{ref}} \,\bigr|,
\qquad
\mathcal{T}_z \;=\; \{\, t : F_{1,z}(t) \ge \alpha \, F_{1,z}(t_{\mathrm{ref}}) \,\}
